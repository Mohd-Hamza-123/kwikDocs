### **1. `pop()`**

Removes the last element from an array and returns it.

    let numbers = [1, 2, 3, 4];
    let last = numbers.pop();
    console.log(numbers); // Output: [1, 2, 3]
    console.log(last);    // Output: 4
    

### **2. `push()`**

Adds one or more elements to the end of an array and returns the new length.

    let numbers = [1, 2, 3];
    let newLength = numbers.push(4, 5);
    console.log(numbers);  // Output: [1, 2, 3, 4, 5]
    console.log(newLength); // Output: 5
    

### **3. `shift()`**

Removes the first element from an array and returns it.

    let numbers = [1, 2, 3, 4];
    let first = numbers.shift();
    console.log(numbers); // Output: [2, 3, 4]
    console.log(first);   // Output: 1
    

### **4. `unshift()`**

Adds one or more elements to the beginning of an array and returns the new length.

    let numbers = [2, 3, 4];
    let newLength = numbers.unshift(0, 1);
    console.log(numbers);  // Output: [0, 1, 2, 3, 4]
    console.log(newLength); // Output: 5
    

### **5. `concat()`**

Merges two or more arrays into a new array.

    let arr1 = [1, 2];
    let arr2 = [3, 4];
    let merged = arr1.concat(arr2, [5, 6]);
    console.log(merged);  // Output: [1, 2, 3, 4, 5, 6]
    console.log(arr1);    // Output: [1, 2] (original array is unchanged)
    

### **6. `indexOf()`**

Returns the index of the first occurrence of a value, or `-1` if not found.

    let numbers = [1, 2, 3, 4, 3];
    console.log(numbers.indexOf(3)); // Output: 2
    console.log(numbers.indexOf(5)); // Output: -1
    

### **7. `includes()`**

Checks if an array contains a specific value. Returns `true` or `false`.

    let fruits = ["apple", "banana", "mango"];
    console.log(fruits.includes("banana")); // Output: true
    console.log(fruits.includes("grape"));  // Output: false
    

### **8. `reverse()`**

Reverses the order of elements in an array.

    let numbers = [1, 2, 3, 4];
    numbers.reverse();
    console.log(numbers); // Output: [4, 3, 2, 1]
    

### **9. `slice()`**

Extracts a section of an array and returns a new array. Does not modify the original array.

    let numbers = [1, 2, 3, 4, 5];
    let sliced = numbers.slice(1, 4); // From index 1 to index 4 (not inclusive)
    console.log(sliced);  // Output: [2, 3, 4]
    console.log(numbers); // Output: [1, 2, 3, 4, 5] (original remains unchanged)
    

### **10. `splice()`**

Adds or removes elements from an array.

#### Remove:

    let numbers = [1, 2, 3, 4, 5];
    let removed = numbers.splice(1, 2); // Remove 2 elements starting at index 1
    console.log(numbers); // Output: [1, 4, 5]
    console.log(removed); // Output: [2, 3]
    

#### Add:

    let numbers = [1, 4, 5];
    numbers.splice(1, 0, 2, 3); // Add 2, 3 at index 1
    console.log(numbers); // Output: [1, 2, 3, 4, 5]
