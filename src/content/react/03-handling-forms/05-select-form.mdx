Handling a `<select>` dropdown with React state is actually **simpler than checkboxes or radios**.
You just store **one value** (for single select) or an **array** (for multi-select).

---

## ðŸ”¹ 1) Single Select Example

```tsx
import { useState } from "react";

function SelectExample() {
  const [language, setLanguage] = useState(""); // single value

  const handleChange = (e: React.ChangeEvent<HTMLSelectElement>) => {
    setLanguage(e.target.value);
  };

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    console.log("Selected language:", language);
  };

  return (
    <form onSubmit={handleSubmit}>
      <h3>Select a Language</h3>
      <select value={language} onChange={handleChange} defaultValue="Something">
        <option value="">-- Choose one --</option>
        <option value="js">JavaScript</option>
        <option value="python">Python</option>
        <option value="java">Java</option>
        <option value="php">PHP</option>
      </select>

      <br /><br />
      <button type="submit">Submit</button>
    </form>
  );
}

export default SelectExample;
```

âœ… If you choose **Python**, state becomes:

```js
"python"
```

## ðŸ”¹ 2) Multi-Select Example

```tsx
import { useState } from "react";

function MultiSelectExample() {
  const [skills, setSkills] = useState<string[]>([]); // multiple values

  const handleChange = (e: React.ChangeEvent<HTMLSelectElement>) => {
    // get selected options as array
    const selectedValues = Array.from(e.target.selectedOptions, (opt) => opt.value);
    setSkills(selectedValues);
  };

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    console.log("Selected skills:", skills);
  };

  return (
    <form onSubmit={handleSubmit}>
      <h3>Select Multiple Skills</h3>
      <select multiple value={skills} onChange={handleChange}>
        <option value="js">JavaScript</option>
        <option value="react">React</option>
        <option value="node">Node.js</option>
        <option value="java">Java</option>
        <option value="php">PHP</option>
      </select>

      <br /><br />
      <button type="submit">Submit</button>
    </form>
  );
}

export default MultiSelectExample;
```

âœ… If you select **React + Node.js**, state becomes:

```js
["react", "node"]
```
