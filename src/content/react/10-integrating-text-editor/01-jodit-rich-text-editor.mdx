Jodit is a free, open-source, and multi-language rich text editor for the web. It is written in JavaScript and uses Vue.js as its framework.

### Install Jodit Editor 

```bash
npm install jodit-react
```

### Import Jodit Editor In Component 

```jsx
import React, { useState } from 'react'
import JoditEditor from 'jodit-react';

const CreatePost = () => {

    const handleSubmit = async (e) => {
   
            e.preventDefault()
            const formData = new FormData(e.target)
            const title = formData.get('title')
            const content = formData.get('content')
            console.log(title, content)
    }

    return (
            <h1>Create Post</h1>
            <form onSubmit={handleSubmit}>
                <input 
                type="text" 
                placeholder="Title" 
                name='title' 
                className='border' 
                />
                <JoditEditor name='content' />
                <button type='submit'>Submit</button>
            </form>
    )
}

export default CreatePost
```

### Configuration 

```jsx 

import React, { useState } from 'react'
import JoditEditor from 'jodit-react';

const config = {
  readonly: false,
  placeholder: 'Write your article here...',
  theme: 'default',
  height: 450,
  toolbarButtonSize: 'middle',
  buttons: [
    'bold','italic','underline','strikethrough','|',
    'brush','eraser','fontsize','font','align','|',
    'ul','ol','outdent','indent','|',
    'link','image','video','table','hr','symbols','|',

    'undo','redo','source'
  ],
  showCharsCounter: true,
  showWordsCounter: true,

  enter: 'p',
  spellcheck: true

    return (
        <JoditEditor config={config} name='content' />
```
